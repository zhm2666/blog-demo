# Gin-Vue-Blog 多人博客改造方案

## 一、改造目标

将现有的**个人博客**改造为**多人博客系统**，实现以下功能：

| 功能 | 改造前 | 改造后 |
|------|--------|--------|
| 浏览文章 | 公开 | 公开 |
| 注册账号 | 仅管理员后台添加 | 用户自主注册 |
| 登录 | 仅管理员 | 用户自主登录 |
| 发表评论 | 需要登录 | 需要登录 |
| 点赞文章/评论 | 需要登录 | 需要登录 |
| 发表文章 | 仅管理员后台 | 登录用户前台投稿 |

// 用户发文章（前台投稿）
func (*Front) UserSaveOrUpdate(c *gin.Context) {
    var req FArticleReq
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    // 获取当前登录用户
    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 检查用户是否被禁用
    if auth.IsDisable {
        ReturnError(c, g.ErrUserDisable, "用户已被禁用")
        return
    }

    // 设置文章状态为"待审核"
    isReview := model.GetConfigBool(db, g.CONFIG_IS_ARTICLE_REVIEW)

    // 保存文章
    article, err := model.SaveUserArticle(db, auth.ID, req, isReview)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, article)
}

// 用户删除自己的文章
func (*Front) UserDeleteArticle(c *gin.Context) {
    var req struct {
        IDs []int `json:"ids"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 只能删除自己的文章
    err = model.DeleteUserArticle(db, req.IDs, auth.ID)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, nil)
}

// 获取用户自己的文章列表
func (*Front) GetUserArticleList(c *gin.Context) {
    var query PageQuery
    if err := c.ShouldBindQuery(&query); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 只查询当前用户的文章
    data, total, err := model.GetUserArticleList(db, query.Page, query.Size, auth.ID)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, PageResult[model.Article]{
        Page:  query.Page,
        Size:  query.Size,
        Total: total,
        List:  data,
    })
}

### 4.3 Model 层方法

**修改文件**: `gin-blog-server/internal/model/article.go`

```go
// 保存用户文章
func SaveUserArticle(db *gorm.DB, userId int, req FArticleReq, isReview bool) (*Article, error) {
    article := Article{
        Title:      req.Title,
        Content:    req.Content,
        CategoryId: req.CategoryId,
        IsTop:      req.IsTop,
        IsReview:   isReview,  // 需审核标记
        Status:     1,
        UserId:     userId,    // 设置作者ID
    }

    // 如果是更新
    if req.ID > 0 {
        var existing Article
        if err := db.Take(&existing, req.ID).Error; err != nil {
            return nil, err
        }
        // 检查文章是否属于该用户
        if existing.UserId != userId {
            return nil, errors.New("无权修改他人的文章")
        }
        article.Model = Model{ID: req.ID}
    }

    result := db.Save(&article)
    return &article, result.Error
}

// 删除用户文章
func DeleteUserArticle(db *gorm.DB, ids []int, userId int) error {
    // 只删除属于该用户的文章
    result := db.Where("id IN ? AND user_id = ?", ids, userId).Delete(&Article{})
    return result.Error
}

// 获取用户文章列表
func GetUserArticleList(db *gorm.DB, page, size, userId int) (list []Article, total int64, err error) {
    query := db.Model(&Article{}).Where("user_id = ?", userId)

    result := query.Count(&total).
        Scopes(Paginate(page, size)).
        Order("created_at DESC").
        Find(&list)

    return list, total, result.Error
}
```

// 后台审核文章
func (a *Article) UpdateReview(c *gin.Context) {
    var req struct {
        IDs      []int `json:"ids"`
        IsReview bool  `json:"is_review"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    db := GetDB(c)
    db.Model(&Article{}).Where("id IN ?", req.IDs).Update("is_review", req.IsReview)

    ReturnSuccess(c, nil)
}
