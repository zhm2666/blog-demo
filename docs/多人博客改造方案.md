# Gin-Vue-Blog 多人博客改造方案

## 一、改造目标

将现有的**个人博客**改造为**多人博客系统**，实现以下功能：

| 功能 | 改造前 | 改造后 |
|------|--------|--------|
| 浏览文章 | 公开 | 公开 |
| 注册账号 | 仅管理员后台添加 | 用户自主注册 |
| 登录 | 仅管理员 | 用户自主登录 |
| 发表评论 | 需要登录 | 游客也可评论（需审核） |
| 点赞文章/评论 | 需要登录 | 需要登录 |
| 发表文章 | 仅管理员后台 | 登录用户前台投稿 |

```go
func (*Front) SaveMessage(c *gin.Context) {
    var req FAddMessageReq
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    // XSS 过滤
    req.Content = template.HTMLEscapeString(req.Content)
    
    db := GetDB(c)
    ipAddress := utils.IP.GetIpAddress(c)
    ipSource := utils.IP.GetIpSource(ipAddress)
    isReview := model.GetConfigBool(db, g.CONFIG_IS_COMMENT_REVIEW)

    var nickname, avatar string
    var userId *int

    // 尝试获取登录用户信息
    auth, err := CurrentUserAuth(c)
    if err == nil && auth != nil {
        // 登录用户
        userId = &auth.ID
        nickname = auth.UserInfo.Nickname
        avatar = auth.UserInfo.Avatar
    } else {
        // 游客 - 使用提交的昵称
        if req.Nickname == "" {
            ReturnError(c, g.ErrRequest, "请输入昵称")
            return
        }
        nickname = template.HTMLEscapeString(req.Nickname)
        
        // 游客默认头像
        avatar = "https://cube.elemecdn.com/3/7c/3ea6beec64369c2642b92c6726f1epng.png"
    }

    // 保存留言
    message, err := model.SaveMessage(db, nickname, avatar, req.Content, 
        ipAddress, ipSource, req.Speed, isReview, userId)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, message)
}
```

#### 4.2.3 用户投稿 Handler

```go
// 用户发文章（前台投稿）
func (*Front) UserSaveOrUpdate(c *gin.Context) {
    var req FArticleReq
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    // 获取当前登录用户
    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 检查用户是否被禁用
    if auth.IsDisable {
        ReturnError(c, g.ErrUserDisable, "用户已被禁用")
        return
    }

    // 设置文章状态为"待审核"
    isReview := model.GetConfigBool(db, g.CONFIG_IS_ARTICLE_REVIEW)

    // 保存文章
    article, err := model.SaveUserArticle(db, auth.ID, req, isReview)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, article)
}

// 用户删除自己的文章
func (*Front) UserDeleteArticle(c *gin.Context) {
    var req struct {
        IDs []int `json:"ids"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 只能删除自己的文章
    err = model.DeleteUserArticle(db, req.IDs, auth.ID)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, nil)
}

// 获取用户自己的文章列表
func (*Front) GetUserArticleList(c *gin.Context) {
    var query PageQuery
    if err := c.ShouldBindQuery(&query); err != nil {
        ReturnError(c, g.ErrRequest, err)
        return
    }

    auth, err := CurrentUserAuth(c)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    db := GetDB(c)

    // 只查询当前用户的文章
    data, total, err := model.GetUserArticleList(db, query.Page, query.Size, auth.ID)
    if err != nil {
        ReturnError(c, g.ErrDbOp, err)
        return
    }

    ReturnSuccess(c, PageResult[model.Article]{
        Page:  query.Page,
        Size:  query.Size,
        Total: total,
        List:  data,
    })
}
```

#### 4.3.3 Article Model 扩展

```go
type Article struct {
    Model
    // ... 现有字段 ...
    
    UserId   int   `json:"user_id" gorm:"index"`  // 投稿用户ID
    IsReview bool  `json:"is_review"`              // 是否待审核
}

// 保存用户文章
func SaveUserArticle(db *gorm.DB, userId int, req FArticleReq, isReview bool) (*Article, error) {
    article := Article{
        Title:      req.Title,
        Content:    req.Content,
        CategoryId: req.CategoryId,
        IsTop:      req.IsTop,
        IsReview:   isReview,
        Status:     1,
        UserId:    userId,
    }

    if req.ID > 0 {
        var existing Article
        if err := db.Take(&existing, req.ID).Error; err != nil {
            return nil, err
        }
        if existing.UserId != userId {
            return nil, errors.New("无权修改他人的文章")
        }
        article.Model = Model{ID: req.ID}
    }

    result := db.Save(&article)
    return &article, result.Error
}
```

### 8.1 游客评论限制

```go
// 限制游客评论频率
func CheckCommentFrequency(c *gin.Context, ipAddress string) bool {
    key := "comment:ip:" + ipAddress
    
    // 检查 Redis 中是否存在
    count, _ := rdb.Get(rctx, key).Int()
    if count >= 5 {
        return false  // 每小时最多评论5次
    }
    
    // 增加计数
    rdb.Incr(rctx, key)
    rdb.Expire(rctx, key, time.Hour)
    
    return true
}
```
