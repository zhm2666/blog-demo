# Gin-Vue-Blog 前后端交互说明文档

## 一、交互概述

Gin-Vue-Blog 采用典型的 **前后端分离** 架构，通过 **HTTP API** 进行数据交互。

```
┌─────────────────────────────────────────────────────────────────────────────┐

└─────────────────────────────────────────────────────────────────────────────┘

   前端 (Vue3 + Axios)                         后端 (Go + Gin)

│                      │    HTTP 请求    │                      │
│   ┌──────────────┐  │ ──────────────▶ │   ┌──────────────┐   │
│   │   Views      │  │                 │   │   Router     │   │
│   │   (页面组件)  │  │                 │   │   (路由匹配)  │   │
│   └──────┬─────────┘  │                 │   └──────┬───────┘   │
│          │              │                 │          │           │
```

### 2.1 完整请求流程

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                         完整请求流程图                                       │
└─────────────────────────────────────────────────────────────────────────────┘

  1. 用户操作
     │
     ▼
  2. 前端发起请求
     │
     │  axios.get('/api/article/list')
     │  axios.post('/api/login', {username, password})
     ▼
  3. Axios 拦截器处理
     │
     │  - 请求拦截器：添加 Token 到 Header
     │  - 响应拦截器：处理错误码、Token 过期
     ▼
  4. HTTP 请求发送
     │
     │  GET /api/article/list HTTP/1.1
     │  Host: localhost:8765
     │  Authorization: Bearer eyJhbGciOiJIUzI1...
     ▼
  5. 后端接收请求
     │
     │  Gin 框架匹配路由
     ▼
  6. 中间件处理 (按顺序)
     │
     │  ① JWTAuth() - 验证 Token
     │  ② PermissionCheck() - 权限验证
     │  ③ OperationLog() - 操作日志
     │  ④ ListenOnline() - 在线状态
     ▼
  7. Handler 处理业务
     │
     │  - 接收请求参数
     │  - 调用 Model 层操作数据库
     │  - 返回响应数据
     ▼
  8. 响应返回
     │
     │  HTTP/1.1 200 OK
     │  Content-Type: application/json
     │  {"code": 0, "message": "OK", "data": [...]}
     ▼
  9. 前端接收响应
     │
     │  - 响应拦截器处理
     │  - 更新页面数据 / 提示错误
     ▼
  10. 页面更新
```

## 三、前端交互实现

### 3.1 Axios 封装

**文件位置**: `gin-blog-admin/src/utils/http.js`

```javascript
import axios from 'axios'
import { useAuthStore } from '@/store'

// 创建 Axios 实例
export const request = axios.create({
    baseURL: import.meta.env.VITE_BASE_API,  // API 基础路径
    timeout: 12000,  // 请求超时时间
})

// 请求拦截器
request.interceptors.request.use(
    (config) => {
        // 不需要 Token 的请求直接放行
        if (config.noNeedToken) {
            return config
        }

        // 从 Pinia 获取 Token
        const { token } = useAuthStore()
        if (token) {
            // 添加到请求头
            config.headers.Authorization = `Bearer ${token}`
        }
        return config
    },
    (error) => Promise.reject(error)
)

// 响应拦截器
request.interceptors.response.use(
    (response) => {
        const { code, message, data } = response.data
        
        // 业务状态码不为 0，表示出错
        if (code !== 0) {
            const authStore = useAuthStore()
            
            // Token 问题，重新登录
            if (code === 1201) {
                authStore.toLogin()
            }
            // Token 过期，强制下线
            if (code === 1202 || code === 1203 || code === 1207) {
                authStore.forceOffline()
            }
            
            return Promise.reject(response.data)
        }
        
        return Promise.resolve(response.data)
    },
    (error) => Promise.reject(error)
)
```

### 4.1 统一响应结构

```go
// 统一的响应结构体
type Response[T any] struct {
    Code    int    `json:"code"`    // 业务状态码 (0=成功)
    Message string `json:"message"` // 响应消息
    Data    T      `json:"data"`    // 响应数据
}

// 成功响应
func ReturnSuccess(c *gin.Context, data any) {
    c.JSON(http.StatusOK, Response[any]{
        Code:    0,           // 0 表示成功
        Message: "OK",
        Data:    data,
    })
}

// 失败响应
func ReturnError(c *gin.Context, r g.Result, data any) {
    c.JSON(http.StatusOK, Response[any]{
        Code:    r.Code(),    // 错误码
        Message: r.Msg(),     // 错误消息
        Data:    data,
    })
}
```

### 4.3 Handler 处理示例

**文件位置**: `gin-blog-server/internal/handle/handle_auth.go`

```go
// 登录处理
func (*UserAuth) Login(c *gin.Context) {
    // 1. 获取请求参数
    var req LoginReq
    if err := c.ShouldBindJSON(&req); err != nil {
        ReturnError(c, g.ErrRequest, err)  // 返回错误
        return
    }

    db := GetDB(c)

    // 2. 查询用户
    userAuth, err := model.GetUserAuthInfoByName(db, req.Username)
    if err != nil {
        ReturnError(c, g.ErrUserNotExist, nil)
        return
    }

    // 3. 验证密码
    if !utils.BcryptCheck(req.Password, userAuth.Password) {
        ReturnError(c, g.ErrPassword, nil)
        return
    }

    // 4. 生成 Token
    roleIds := []int{}
    for _, role := range userAuth.Roles {
        roleIds = append(roleIds, role.ID)
    }
    token, _ := jwt.GenToken(g.Conf.JWT.Secret, g.Conf.JWT.Issuer,
        g.Conf.JWT.Expire, userAuth.ID, roleIds)

    // 5. 返回成功响应
    ReturnSuccess(c, LoginVO{
        UserInfo: *userAuth.UserInfo,
        Token:    token,
    })
}
```

```go
func registerAdminHandler(r *gin.Engine) {
```

### 6.2 前端 Token 存储

```javascript
// 使用 Pinia 管理 Token
// 文件: src/store/modules/auth.js

import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useAuthStore = defineStore('auth', () => {
    const token = ref(localStorage.getItem('token') || '')
    
    const setToken = (newToken) => {
        token.value = newToken
        localStorage.setItem('token', newToken)
    }
    
    const removeToken = () => {
        token.value = ''
        localStorage.removeItem('token')
    }
    
    return { token, setToken, removeToken }
})
```

### 6.3 后端 Token 验证

```go
// JWT 中间件
func JWTAuth() gin.HandlerFunc {
    return func(c *gin.Context) {
        // 1. 从 Header 获取 Token
        auth := c.Request.Header.Get("Authorization")
        
        // 2. 解析 Bearer Token
        parts := strings.Split(auth, " ")
        if len(parts) != 2 || parts[0] != "Bearer" {
            ReturnError(c, g.ErrTokenType, nil)
            return
        }
        
        // 3. 验证 Token
        claims, err := jwt.ParseToken(secret, parts[1])
        if err != nil {
            ReturnError(c, g.ErrTokenWrong, err)
            return
        }
        
        // 4. 检查过期
        if time.Now().Unix() > claims.ExpiresAt.Unix() {
            ReturnError(c, g.ErrTokenRuntime, nil)
            return
        }
        
        // 5. 获取用户信息并存入 Context
        user, _ := model.GetUserAuthInfoById(db, claims.UserId)
        c.Set("user_auth", user)
    }
}
```

## 八、错误处理

### 8.1 前端错误处理

```javascript
// 响应拦截器中的错误处理
request.interceptors.response.use(
    (response) => {
        const { code, message, data } = response.data
        
        if (code !== 0) {
            // 业务错误
            window.$message.error(message)
            
            // 特殊错误码处理
            if (code === 1201 || code === 1202) {
                // Token 问题，重新登录
                authStore.toLogin()
            }
            
            return Promise.reject(response.data)
        }
        
        return Promise.resolve(response.data)
    },
    (error) => {
        // HTTP 错误 (如 500)
        if (error.response?.status === 500) {
            window.$message.error('服务端异常')
        }
        return Promise.reject(error)
    }
)
```

### 8.2 后端错误处理

```go
// 统一错误响应
func ReturnError(c *gin.Context, r g.Result, data any) {

        Data:    data,

func (*UserAuth) Login(c *gin.Context) {
    var req LoginReq

}
```

```
│  3. 状态码设计                                                              │
│     - code = 0: 成功                                                        │
│     - code > 0: 业务错误 (1201/1202/1206 等)                               │
│     - HTTP 500: 系统错误                                                    │
│                                                                             │
│  4. 请求库: Axios                                                          │
│     - 请求/响应拦截器                                                       │

│     - 错误统一处理                                                          │
│                                                                             │
│  5. 后端框架: Gin                                                          │
│     - 中间件链: JWT → 权限 → 日志 → Handler                               │
│     - 统一响应格式                                                          │
│     - RESTful API 风格                                                     │
│                                                                             │
```
